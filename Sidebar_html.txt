    <script>
      function runSearch() {
        var classNum = document.getElementById('classNum').value;
        var studentId = document.getElementById('studentId').value;
        var resultDiv = document.getElementById('result');
        
        if (!classNum || !studentId) {
          resultDiv.style.display = 'block';
          resultDiv.innerHTML = '<p class="error">이동반과 학번을 모두 입력해주세요.</p>';
          return;
        }
        
        resultDiv.style.display = 'block';
        resultDiv.innerHTML = '<p style="text-align:center; color:#666;">조회 중...</p>';
        
        // 서버 측(Code.gs) 함수 호출
        google.script.run
          .withSuccessHandler(function(response) {
            if (response.success) {
              var s = response.data;
              var html = '<h4>조회 결과</h4>' +
                '<div class="result-row"><span>이동반</span><span>' + s.classNum + '반</span></div>' +
                '<div class="result-row"><span>학번</span><span>' + s.studentId + '</span></div>' +
                '<div class="result-row"><span>이름</span><span>' + s.name + '</span></div>' +
                '<div class="result-row"><span>수행 1차</span><span>' + s.perf1 + '점</span></div>' +
                '<div class="result-row"><span>수행 2차</span><span>' + s.perf2 + '점</span></div>' +
                '<div class="result-row"><span>수행 3차</span><span>' + s.perf3 + '점</span></div>' +
                '<div class="result-row"><span>수행 (합계)</span><span>' + s.total + '점</span></div>';
              resultDiv.innerHTML = html;
            } else {
              resultDiv.innerHTML = '<p class="error">' + response.message + '</p>';
            }
          })
          .searchStudent(classNum, studentId);
      }
    </script>
